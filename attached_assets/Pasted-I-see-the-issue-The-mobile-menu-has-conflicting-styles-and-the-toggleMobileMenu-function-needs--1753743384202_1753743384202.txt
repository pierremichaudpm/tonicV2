```javascript
<script>
/* Fix 3: Define toggleMobileMenu function */
function toggleMobileMenu() {
    var menu = document.getElementById('mobileMenu');
    var button = document.getElementById('mobileMenuButton');
    
    if (!menu) return;
    
    if (menu.classList.contains('active')) {
        // Close menu
        menu.classList.remove('active');
        menu.style.display = 'none';
        document.body.style.overflow = '';
        
        // Reset hamburger icon
        if (button) {
            const svg = button.querySelector('svg');
            if (svg) {
                svg.innerHTML = '<path stroke-linecap="round" stroke-linejoin="round" d="M4 6h16M4 12h16M4 18h16"></path>';
            }
        }
    } else {
        // Open menu
        menu.classList.add('active');
        menu.style.display = 'flex';
        document.body.style.overflow = 'hidden';
        
        // Change to X icon
        if (button) {
            const svg = button.querySelector('svg');
            if (svg) {
                svg.innerHTML = '<path d="M6 18L18 6M6 6l12 12"></path>';
            }
        }
    }
}
</script>
```
